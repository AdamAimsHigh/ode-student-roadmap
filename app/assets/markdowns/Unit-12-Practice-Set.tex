\section*{Part II --- Review Guide: Practice Problems}

Work the following twelve problems in order. They climb through the defining integral,
linearity and the table, the first shift, the derivative rule on a full IVP, the
integral/$t$-multiplication rules, inverse transforms by partial fractions and by
completing the square, a repeated root, a complete five-step solution, Heaviside
shifting forward and back, and a Dirac-delta impulse-response capstone. Solutions
follow in Part III.

\begin{enumerate}[leftmargin=2em]
    \item \textbf{(From the definition)} Using only the defining integral
          $\int_0^\infty e^{-st}f\,dt$, compute $\mathcal{L}\{e^{3t}\}$ and state the
          half-plane of convergence.

    \item \textbf{(Linearity and the table)} Find
          $\mathcal{L}\{3t^2 - 2\cos 4t + 5\}$.

    \item \textbf{(First shift theorem)} Find $\mathcal{L}\{e^{-2t}\sin 3t\}$.

    \item \textbf{(Derivative rule on an IVP)} Transform $y'' + 4y = 0$ with
          $y(0)=1$, $y'(0)=0$, solve for $Y(s)$, and invert to find $y(t)$.

    \item \textbf{($t$-multiplication rule)} Find $\mathcal{L}\{t\,e^{2t}\}$ using
          $\mathcal{L}\{t f(t)\} = -F'(s)$.

    \item \textbf{(Inverse --- distinct roots)} Find
          $\mathcal{L}^{-1}\!\left\{\dfrac{s+7}{(s-1)(s+3)}\right\}$.

    \item \textbf{(Inverse --- complete the square)} Find
          $\mathcal{L}^{-1}\!\left\{\dfrac{1}{s^2+4s+13}\right\}$.

    \item \textbf{(Inverse --- repeated root)} Find
          $\mathcal{L}^{-1}\!\left\{\dfrac{3s+1}{(s+2)^2}\right\}$.

    \item \textbf{(Full five-step IVP)} Solve $y'' + 3y' + 2y = 0$ with $y(0)=1$,
          $y'(0)=0$ by the Laplace method.

    \item \textbf{(Heaviside shifting, forward)} Find
          $\mathcal{L}\{u(t-3)\cos(t-3)\}$, then find
          $\mathcal{L}\{u(t-2)\,t^2\}$ by first re-expressing $t^2$ in $(t-2)$.

    \item \textbf{(Heaviside shifting, inverse)} Find
          $\mathcal{L}^{-1}\!\left\{\dfrac{e^{-2s}}{s^2}\right\}$.

    \item \textbf{(Dirac impulse response, capstone)} Solve $y'' + 4y = \delta(t-\pi)$
          with $y(0)=0$, $y'(0)=0$, and describe the motion before and after the kick.
\end{enumerate}

\section*{Part III --- Complete Solutions}

\noindent\textbf{Solution 1.}\quad Merge the two exponentials and integrate:
\[ \textcolor{workpurple}{\int_0^\infty e^{-st}e^{3t}\,dt = \int_0^\infty e^{-(s-3)t}\,dt
   = \left[-\tfrac{1}{s-3}e^{-(s-3)t}\right]_0^\infty = \frac{1}{s-3}.} \]
The boundary term at $\infty$ vanishes only when $s-3>0$, so the transform converges
for $s>3$.

\medskip
\noindent\textbf{Solution 2.}\quad Apply linearity term by term with
$\mathcal{L}\{t^2\}=2/s^3$, $\mathcal{L}\{\cos 4t\}=s/(s^2+16)$, $\mathcal{L}\{1\}=1/s$:
\[ \textcolor{workpurple}{\mathcal{L}\{3t^2 - 2\cos 4t + 5\}
   = \frac{6}{s^3} - \frac{2s}{s^2+16} + \frac{5}{s}.} \]

\medskip
\noindent\textbf{Solution 3.}\quad Start from $\mathcal{L}\{\sin 3t\}=\dfrac{3}{s^2+9}$
and apply the first shift $s\mapsto s+2$:
\[ \textcolor{workpurple}{\mathcal{L}\{e^{-2t}\sin 3t\}
   = \left.\frac{3}{s^2+9}\right|_{s\to s+2} = \frac{3}{(s+2)^2+9}.} \]
The completed-square denominator $(s+2)^2+9$ is the fingerprint of a damped oscillation.

\medskip
\noindent\textbf{Solution 4.}\quad Transform with $\mathcal{L}\{y''\}=s^2Y-sy(0)-y'(0)$
and substitute $y(0)=1$, $y'(0)=0$:
\[ \textcolor{workpurple}{(s^2 Y - s) + 4Y = 0 \;\Rightarrow\; (s^2+4)Y = s
   \;\Rightarrow\; Y(s) = \frac{s}{s^2+4}.} \]
This is exactly $\mathcal{L}\{\cos 2t\}$, so
\[ \textcolor{workpurple}{y(t) = \cos 2t.} \]

\medskip
\noindent\textbf{Solution 5.}\quad With $F(s)=\mathcal{L}\{e^{2t}\}=\dfrac{1}{s-2}$,
apply $\mathcal{L}\{tf\}=-F'(s)$:
\[ \textcolor{workpurple}{\mathcal{L}\{t e^{2t}\} = -\frac{d}{ds}\!\left(\frac{1}{s-2}\right)
   = -\left(-\frac{1}{(s-2)^2}\right) = \frac{1}{(s-2)^2}.} \]
Equivalently, this is the first shift applied to $\mathcal{L}\{t\}=1/s^2$.

\medskip
\noindent\textbf{Solution 6.}\quad Decompose
$\dfrac{s+7}{(s-1)(s+3)} = \dfrac{A}{s-1} + \dfrac{B}{s+3}$. Clearing denominators,
$s+7 = A(s+3) + B(s-1)$. Set $s=1$: $8=4A\Rightarrow A=2$. Set $s=-3$:
$4=-4B\Rightarrow B=-1$:
\[ \textcolor{workpurple}{\frac{s+7}{(s-1)(s+3)} = \frac{2}{s-1} - \frac{1}{s+3}
   \;\Longrightarrow\; \mathcal{L}^{-1} = 2e^{t} - e^{-3t}.} \]

\medskip
\noindent\textbf{Solution 7.}\quad Complete the square in the denominator:
$s^2+4s+13 = (s+2)^2 + 9$. This matches the shifted $\sin$ with $a=-2$, $\omega=3$,
and a $1/\omega$ scale:
\[ \textcolor{workpurple}{\frac{1}{(s+2)^2+3^2}
   = \frac{1}{3}\cdot\frac{3}{(s+2)^2+3^2}
   \;\Longrightarrow\; \mathcal{L}^{-1} = \frac{1}{3}\,e^{-2t}\sin 3t.} \]

\medskip
\noindent\textbf{Solution 8.}\quad A repeated factor needs two terms,
$\dfrac{3s+1}{(s+2)^2} = \dfrac{A}{s+2} + \dfrac{B}{(s+2)^2}$. Then
$3s+1 = A(s+2)+B$, so $A=3$ and $B = 1-2A = -5$:
\[ \textcolor{workpurple}{\frac{3}{s+2} - \frac{5}{(s+2)^2}
   \;\Longrightarrow\; \mathcal{L}^{-1} = 3e^{-2t} - 5t\,e^{-2t} = (3-5t)\,e^{-2t}.} \]
The $t e^{-2t}$ term is the signature of the repeated real root.

\medskip
\noindent\textbf{Solution 9.}\quad Transform and substitute $y(0)=1$, $y'(0)=0$:
\[ \textcolor{workpurple}{(s^2 Y - s) + 3(sY - 1) + 2Y = 0
   \;\Rightarrow\; (s^2+3s+2)Y = s+3.} \]
Factor $s^2+3s+2=(s+1)(s+2)$ and decompose
$\dfrac{s+3}{(s+1)(s+2)} = \dfrac{A}{s+1}+\dfrac{B}{s+2}$ with $A=2$, $B=-1$:
\[ \textcolor{workpurple}{Y(s) = \frac{2}{s+1} - \frac{1}{s+2}
   \;\Longrightarrow\; y(t) = 2e^{-t} - e^{-2t}.} \]

\medskip
\noindent\textbf{Solution 10.}\quad The first part is the second shift applied to
$\mathcal{L}\{\cos t\}=\dfrac{s}{s^2+1}$ with $c=3$:
\[ \textcolor{workpurple}{\mathcal{L}\{u(t-3)\cos(t-3)\} = e^{-3s}\,\frac{s}{s^2+1}.} \]
For the second part, the argument $t^2$ must be rewritten in $(t-2)$ before the
theorem applies: $t^2 = (t-2)^2 + 4(t-2) + 4$. Transform each shifted piece
($\mathcal{L}\{t^2\}=2/s^3$, $\mathcal{L}\{t\}=1/s^2$, $\mathcal{L}\{1\}=1/s$) and
attach $e^{-2s}$:
\[ \textcolor{workpurple}{\mathcal{L}\{u(t-2)\,t^2\}
   = e^{-2s}\!\left(\frac{2}{s^3} + \frac{4}{s^2} + \frac{4}{s}\right).} \]

\medskip
\noindent\textbf{Solution 11.}\quad The factor $e^{-2s}$ signals a delay by $c=2$.
Since $\mathcal{L}^{-1}\{1/s^2\}=t$, the second shift gives
\[ \textcolor{workpurple}{\mathcal{L}^{-1}\!\left\{\frac{e^{-2s}}{s^2}\right\}
   = u(t-2)\,(t-2).} \]
The response is a ramp that stays flat at zero until it switches on at $t=2$.

\medskip
\noindent\textbf{Solution 12.}\quad Transform with $\mathcal{L}\{\delta(t-\pi)\}=e^{-\pi s}$
and zero initial data:
\[ \textcolor{workpurple}{(s^2+4)Y = e^{-\pi s}
   \;\Rightarrow\; Y(s) = e^{-\pi s}\,\frac{1}{s^2+4}.} \]
Since $\mathcal{L}^{-1}\{1/(s^2+4)\} = \tfrac{1}{2}\sin 2t$, the second shift gives
\[ \textcolor{workpurple}{y(t) = \frac{1}{2}\,u(t-\pi)\,\sin\big(2(t-\pi)\big)
   = \frac{1}{2}\,u(t-\pi)\,\sin 2t.} \]
The mass sits at rest until the impulse at $t=\pi$, which jolts the velocity and
launches free oscillation thereafter --- the impulse response switched on at the kick.

\begin{conceptbox}[Unit Key Takeaway]
Every problem above is the same loop: \textcolor{accentorange}{transform, do the
algebra, invert}. The derivative rule carries the \textcolor{accentorange}{initial
data} into the equation, completing the square handles complex roots, $e^{-cs}$ tracks
every \textcolor{accentorange}{delay}, and $\mathcal{L}\{\delta\}=e^{-cs}$ turns an
impulse into a clean algebraic term.
\end{conceptbox}
